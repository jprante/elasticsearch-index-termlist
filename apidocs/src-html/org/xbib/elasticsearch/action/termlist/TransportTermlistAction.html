074    @Override
075    protected TermlistResponse newResponse(TermlistRequest request, AtomicReferenceArray shardsResponses, ClusterState clusterState) {
076        int successfulShards = 0;
077        int failedShards = 0;
078        List<ShardOperationFailedException> shardFailures = null;
079        int numdocs = 0;
080        Map<String, TermInfo> map = new CompactHashMap<String, TermInfo>();
081        for (int i = 0; i < shardsResponses.length(); i++) {
082            Object shardResponse = shardsResponses.get(i);
083            if (shardResponse instanceof BroadcastShardOperationFailedException) {
084                BroadcastShardOperationFailedException e = (BroadcastShardOperationFailedException)shardResponse;
085                logger.error(e.getMessage(), e);
086                failedShards++;
087                if (shardFailures == null) {
088                    shardFailures = newLinkedList();
089                }
090                shardFailures.add(new DefaultShardOperationFailedException(e));
091            } else {
092                if (shardResponse instanceof ShardTermlistResponse) {
093                    successfulShards++;
094                    ShardTermlistResponse resp = (ShardTermlistResponse) shardResponse;
095                    numdocs += resp.getNumDocs();
096                    update(map, resp.getTermList());
097                }
098            }
099        }
100        map = request.sortByTotalFreq() ? sortTotalFreq(map, request.getFrom(), request.getSize()) :
101                request.sortByDocFreq() ? sortDocFreq(map, request.getFrom(), request.getSize()) :
102                        request.sortByTerm() ? sortTerm(map, request.getFrom(), request.getSize()) :
103                                request.getSize() >= 0 ? truncate(map, request.getFrom(), request.getSize()) : map;
104
105        return new TermlistResponse(shardsResponses.length(), successfulShards, failedShards, shardFailures, numdocs, map);
106    }

141    @Override
142    protected ShardTermlistResponse shardOperation(ShardTermlistRequest request) throws ElasticsearchException {
143        IndexShard indexShard = indicesService.indexServiceSafe(request.getIndex()).shardSafe(request.shardId().id());
144        Engine.Searcher searcher = indexShard.engine().acquireSearcher("termlist");
145        try {
146            Map<String, TermInfo> map = new CompactHashMap<String, TermInfo>();
147            IndexReader reader = searcher.reader();
148            Fields fields = MultiFields.getFields(reader);
149            if (fields != null) {
150                for (String field : fields) {
151                    // skip internal fields
152                    if (field.charAt(0) == '_') {
153                        continue;
154                    }
155                    if (request.getRequest().getField() == null || field.equals(request.getRequest().getField())) {
156                        Terms terms = fields.terms(field);
157                        // Returns the number of documents that have at least one
158                        if (terms != null) {
159                            TermsEnum termsEnum = terms.iterator(null);
160                            BytesRef text;
161                            while ((text = termsEnum.next()) != null) {
162                                // skip invalid terms
163                                if (termsEnum.docFreq() < 1) {
164                                    continue;
165                                }
166                                if (termsEnum.totalTermFreq() < 1) {
167                                    continue;
168                                }
169                                // docFreq() = the number of documents containing the current term
170                                // totalTermFreq() = total number of occurrences of this term across all documents
171                                Term term = new Term(field, text);
172                                if (request.getRequest().getTerm() == null || term.text().startsWith(request.getRequest().getTerm())) {
173                                    TermInfo termInfo = new TermInfo();
174                                    DocsAndPositionsEnum docPosEnum = termsEnum.docsAndPositions(null, null);
175                                    SummaryStatistics stat = new SummaryStatistics();
176                                    while (docPosEnum.nextDoc() != DocsAndPositionsEnum.NO_MORE_DOCS) {
177                                        stat.addValue(docPosEnum.freq());
178                                    }
179                                    termInfo.setSummaryStatistics(stat);
180                                    termInfo.setDocFreq(termsEnum.docFreq());
181                                    termInfo.setTotalFreq(termsEnum.totalTermFreq());
182                                    map.put(term.text(), termInfo);
183                                }
184                            }
185                        }
186                    }
187                }
188            }
189            return new ShardTermlistResponse(request.getIndex(), request.shardId(), reader.numDocs(), map);
190        } catch (Throwable ex) {
191            logger.error(ex.getMessage(), ex);
192            throw new ElasticsearchException(ex.getMessage(), ex);
193        } finally {
194            searcher.close();
195        }
196    }

198    private void update(Map<String, TermInfo> map, Map<String, TermInfo> other) {
199        for (Map.Entry<String, TermInfo> t2 : other.entrySet()) {
200            if (map.containsKey(t2.getKey())) {
201                TermInfo t1 = map.get(t2.getKey());
202                Long totalFreq = t1.getTotalFreq();
203                if (totalFreq != null) {
204                    if (t2.getValue().getTotalFreq() != null) {
205                        t1.setTotalFreq(totalFreq + t2.getValue().getTotalFreq());
206                    }
207                } else {
208                    if (t2.getValue().getTotalFreq() != null) {
209                        t1.setTotalFreq(t2.getValue().getTotalFreq());
210                    }
211                }
212                Integer docFreq = t1.getDocFreq();
213                if (docFreq != null) {
214                    if (t2.getValue().getDocFreq() != null) {
215                        t1.setDocFreq(docFreq + t2.getValue().getDocFreq());
216                    }
217                } else {
218                    if (t2.getValue().getDocFreq() != null) {
219                        t1.setDocFreq(t2.getValue().getDocFreq());
220                    }
221                }
222                SummaryStatistics summaryStatistics = t1.getSummaryStatistics();
223                if (summaryStatistics != null) {
224                    if (t2.getValue().getSummaryStatistics() != null) {
225                        summaryStatistics.update(t2.getValue().getSummaryStatistics());
226                    }
227                } else {
228                    if (t2.getValue().getSummaryStatistics() != null) {
229                        t1.setSummaryStatistics(t2.getValue().getSummaryStatistics());
230                    }
231                }
232                map.put(t2.getKey(), t1);
233            } else {
234                map.put(t2.getKey(), t2.getValue());
235            }
236        }
237    }

239    private SortedMap<String, TermInfo> sortTerm(final Map<String, TermInfo> map, Integer from, Integer size) {
240        Comparator<String> comp = new Comparator<String>() {
241            @Override
242            public int compare(String t1, String t2) {
243                return t1.compareTo(t2);
244            }
245        };
246        TreeMap<String, TermInfo> m = new TreeMap<String, TermInfo>(comp);
247        m.putAll(map);
248        if (size != null && size > 0) {
249            TreeMap<String, TermInfo> treeMap = new TreeMap<String, TermInfo>(comp);
250            for (int i = 0; i < m.size(); i++) {
251                Map.Entry<String, TermInfo> me = m.pollFirstEntry();
252                if (from <= i && i < from + size) {
253                    treeMap.put(me.getKey(), me.getValue());
254                }
255            }
256            return treeMap;
257        }
258        return m;
259    }
260    private SortedMap<String, TermInfo> sortTotalFreq(final Map<String, TermInfo> map, Integer from, Integer size) {
261        Comparator<String> comp = new Comparator<String>() {
262            @Override
263            public int compare(String t1, String t2) {
264                Long l1 = map.get(t1).getTotalFreq();
265                String sl1 = Long.toString(l1);
266                String s1 = sl1.length() + sl1 + t1;
267                Long l2 = map.get(t2).getTotalFreq();
268                String sl2 = Long.toString(l2);
269                String s2 =sl2.length() + sl2 + t2;
270                return -s1.compareTo(s2);
271            }
272        };
273        TreeMap<String, TermInfo> m = new TreeMap<String, TermInfo>(comp);
274        m.putAll(map);
275        if (size != null && size > 0) {
276            TreeMap<String, TermInfo> treeMap = new TreeMap<String, TermInfo>(comp);
277            for (int i = 0; i < m.size(); i++) {
278                Map.Entry<String, TermInfo> me = m.pollFirstEntry();
279                if (from <= i && i < from + size) {
280                    treeMap.put(me.getKey(), me.getValue());
281                }
282            }
283            return treeMap;
284        }
285        return m;
286    }

288    private SortedMap<String, TermInfo> sortDocFreq(final Map<String, TermInfo> map, Integer from, Integer size) {
289        Comparator<String> comp = new Comparator<String>() {
290            @Override
291            public int compare(String t1, String t2) {
292                Integer i1 = map.get(t1).getDocFreq();
293                String si1 = Integer.toString(i1);
294                String s1 = si1.length() + si1 + t1;
295                Integer i2 = map.get(t2).getDocFreq();
296                String si2 = Integer.toString(i2);
297                String s2 = si2.length() + si2 + t2;
298                return -s1.compareTo(s2);
299            }
300        };
301        TreeMap<String, TermInfo> m = new TreeMap<String, TermInfo>(comp);
302        m.putAll(map);
303        if (size != null && size > 0) {
304            TreeMap<String, TermInfo> treeMap = new TreeMap<String, TermInfo>(comp);
305            for (int i = 0; i < m.size(); i++) {
306                Map.Entry<String, TermInfo> me = m.pollFirstEntry();
307                if (from <= i && i < from + size) {
308                    treeMap.put(me.getKey(), me.getValue());
309                }
310            }
311            return treeMap;
312        }
313        return m;
314    }

316    private Map<String, TermInfo> truncate(Map<String, TermInfo> source, Integer from, Integer size) {
317        if (size == null || size < 1) {
318            return source;
319        }
320        Map<String, TermInfo> target = new CompactHashMap<String, TermInfo>();
321        Iterator<Map.Entry<String, TermInfo>> it = source.entrySet().iterator();
322        for (int i = 0 ; i < source.size(); i++) {
323            Map.Entry<String, TermInfo> entry = it.next();
324            if (from <= i && i < from + size) {
325                target.put(entry.getKey(), entry.getValue());
326            }
327        }
328        return target;
329    }
