003    import java.util.AbstractCollection;

011    /**
012     * A memory-efficient hash map.
013     *
014     * Modified version of com.google.gwt.dev.util.collect.HashMap
015     *
016     * @param <K> the key type
017     * @param <V> the value type
018     */
019    public class CompactHashMap<K, V> implements Map<K, V> {
020    
021        /**
022         * In the interest of memory-savings, we start with the smallest feasible
023         * power-of-two table size that can hold three items without rehashing. If we
024         * started with a size of 2, we'd have to expand as soon as the second item
025         * was added.
026         */
027        private static final int INITIAL_TABLE_SIZE = 4;
028    
029        private static final Object NULL_KEY = new Object() {
030    
031            Object readResolve() {
032                return NULL_KEY;
033            }
034        };
035    
036        static Object maskNullKey(Object k) {
037            return (k == null) ? NULL_KEY : k;
038        }
039    
040        static Object unmaskNullKey(Object k) {
041            return (k == NULL_KEY) ? null : k;
042        }
043        /**
044         * Backing store for all the keys; transient due to custom serialization.
045         * Default access to avoid synthetic accessors from inner classes.
046         */
047        transient Object[] keys;
048        /**
049         * Number of pairs in this set; transient due to custom serialization. Default
050         * access to avoid synthetic accessors from inner classes.
051         */
052        transient int size = 0;
053        /**
054         * Backing store for all the values; transient due to custom serialization.
055         * Default access to avoid synthetic accessors from inner classes.
056         */
057        transient Object[] values;
058    
059        public CompactHashMap() {
060            initTable(INITIAL_TABLE_SIZE);
061        }
062    
063        public CompactHashMap(Map<? extends K, ? extends V> m) {
064            int newCapacity = INITIAL_TABLE_SIZE;
065            int expectedSize = m.size();
066            while (newCapacity * 3 < expectedSize * 4) {
067                newCapacity <<= 1;
068            }
069            initTable(newCapacity);
070            internalPutAll(m);
071        }
072    
073        @Override
074        public void clear() {
075            initTable(INITIAL_TABLE_SIZE);
076            size = 0;
077        }
078    
079        @Override
080        public boolean containsKey(Object key) {
081            return findKey(key) >= 0;
082        }
083    
084        @Override
085        public boolean containsValue(Object value) {
086            if (value == null) {
087                for (int i = 0; i < keys.length; ++i) {
088                    if (keys[i] != null && values[i] == null) {
089                        return true;
090                    }
091                }
092            } else {
093                for (Object existing : values) {
094                    if (valueEquals(existing, value)) {
095                        return true;
096                    }
097                }
098            }
099            return false;
100        }
101    
102        @Override
103        public Set<Entry<K, V>> entrySet() {
104            return new EntrySet();
105        }
106    
107        @Override
108        public boolean equals(Object o) {
109            if (!(o instanceof Map)) {
110                return false;
111            }
112            Map<K, V> other = (Map<K, V>) o;
113            return entrySet().equals(other.entrySet());
114        }
115    
116        @Override
117        public V get(Object key) {
118            int index = findKey(key);
119            return (index < 0) ? null : (V) values[index];
120        }
121    
122        @Override
123        public int hashCode() {
124            int result = 0;
125            for (int i = 0; i < keys.length; ++i) {
126                Object key = keys[i];
127                if (key != null) {
128                    result += keyHashCode(unmaskNullKey(key)) ^ valueHashCode(values[i]);
129                }
130            }
131            return result;
132        }
133    
134        @Override
135        public boolean isEmpty() {
136            return size == 0;
137        }
138    
139        @Override
140        public Set<K> keySet() {
141            return new KeySet();
142        }
143    
144        @Override
145        public V put(K key, V value) {
146            ensureSizeFor(size + 1);
147            int index = findKeyOrEmpty(key);
148            if (keys[index] == null) {
149                ++size;
150                keys[index] = maskNullKey(key);
151                values[index] = value;
152                return null;
153            } else {
154                Object previousValue = values[index];
155                values[index] = value;
156                return (V) previousValue;
157            }
158        }
159    
160        @Override
161        public void putAll(Map<? extends K, ? extends V> m) {
162            ensureSizeFor(size + m.size());
163            internalPutAll(m);
164        }
165    
166        @Override
167        public V remove(Object key) {
168            int index = findKey(key);
169            if (index < 0) {
170                return null;
171            }
172            Object previousValue = values[index];
173            internalRemove(index);
174            return (V) previousValue;
175        }
176    
177        @Override
178        public int size() {
179            return size;
180        }
181    
182        @Override
183        public String toString() {
184            if (size == 0) {
185                return "{}";
186            }
187            StringBuilder buf = new StringBuilder(32 * size());
188            buf.append('{');
189            boolean needComma = false;
190            for (int i = 0; i < keys.length; ++i) {
191                Object key = keys[i];
192                if (key != null) {
193                    if (needComma) {
194                        buf.append(',').append(' ');
195                    }
196                    key = unmaskNullKey(key);
197                    Object value = values[i];
198                    buf.append(key == this ? "(this Map)" : key).append('=').append(
199                            value == this ? "(this Map)" : value);
200                    needComma = true;
201                }
202            }
203            buf.append('}');
204            return buf.toString();
205        }

207        @Override
208        public Collection<V> values() {
209            return new Values();
210        }

212        private class EntryIterator implements Iterator<Entry<K, V>> {
213    
214            private int index = 0;
215            private int last = -1;
216    
217            {
218                advanceToItem();
219            }
220    
221            @Override
222            public boolean hasNext() {
223                return index < keys.length;
224            }
225    
226            @Override
227            public Entry<K, V> next() {
228                if (!hasNext()) {
229                    throw new NoSuchElementException();
230                }
231                last = index;
232                Entry<K, V> toReturn = new HashEntry(index++);
233                advanceToItem();
234                return toReturn;
235            }
236    
237            @Override
238            public void remove() {
239                if (last < 0) {
240                    throw new IllegalStateException();
241                }
242                internalRemove(last);
243                if (keys[last] != null) {
244                    index = last;
245                }
246                last = -1;
247            }
248    
249            private void advanceToItem() {
250                for (; index < keys.length; ++index) {
251                    if (keys[index] != null) {
252                        return;
253                    }
254                }
255            }
256        }

258        private class EntrySet extends AbstractSet<Entry<K, V>> {
259    
260            @Override
261            public boolean add(Entry<K, V> entry) {
262                boolean result = !CompactHashMap.this.containsKey(entry.getKey());
263                CompactHashMap.this.put(entry.getKey(), entry.getValue());
264                return result;
265            }
266    
267            @Override
268            public boolean addAll(Collection<? extends Entry<K, V>> c) {
269                CompactHashMap.this.ensureSizeFor(size() + c.size());
270                return super.addAll(c);
271            }
272    
273            @Override
274            public void clear() {
275                CompactHashMap.this.clear();
276            }
277    
278            @Override
279            public boolean contains(Object o) {
280                if (!(o instanceof Entry)) {
281                    return false;
282                }
283                Entry<K, V> entry = (Entry<K, V>) o;
284                V value = CompactHashMap.this.get(entry.getKey());
285                return CompactHashMap.this.valueEquals(value, entry.getValue());
286            }
287    
288            @Override
289            public int hashCode() {
290                return CompactHashMap.this.hashCode();
291            }
292    
293            @Override
294            public Iterator<java.util.Map.Entry<K, V>> iterator() {
295                return new EntryIterator();
296            }
297    
298            @Override
299            public boolean remove(Object o) {
300                if (!(o instanceof Entry)) {
301                    return false;
302                }
303                Entry<K, V> entry = (Entry<K, V>) o;
304                int index = findKey(entry.getKey());
305                if (index >= 0 && valueEquals(values[index], entry.getValue())) {
306                    internalRemove(index);
307                    return true;
308                }
309                return false;
310            }
311    
312            @Override
313            public boolean removeAll(Collection<?> c) {
314                boolean didRemove = false;
315                for (Object o : c) {
316                    didRemove |= remove(o);
317                }
318                return didRemove;
319            }
320    
321            @Override
322            public int size() {
323                return CompactHashMap.this.size;
324            }
325    
326            @Override
327            public boolean equals(Object obj) {
328                if (obj == null) {
329                    return false;
330                }
331                if (getClass() != obj.getClass()) {
332                    return false;
333                }
334                final EntrySet other = (EntrySet) obj;
335                return true;
336            }
337        }

339        private class HashEntry implements Entry<K, V> {
340    
341            private final int index;
342    
343            public HashEntry(int index) {
344                this.index = index;
345            }
346    
347            @Override
348            public boolean equals(Object o) {
349                if (!(o instanceof Entry)) {
350                    return false;
351                }
352                Entry<K, V> entry = (Entry<K, V>) o;
353                return keyEquals(getKey(), entry.getKey())
354                        && valueEquals(getValue(), entry.getValue());
355            }
356    
357            @Override
358            public K getKey() {
359                return (K) unmaskNullKey(keys[index]);
360            }
361    
362            @Override
363            public V getValue() {
364                return (V) values[index];
365            }
366    
367            @Override
368            public int hashCode() {
369                return keyHashCode(getKey()) ^ valueHashCode(getValue());
370            }
371    
372            @Override
373            public V setValue(V value) {
374                V previous = (V) values[index];
375                values[index] = value;
376                return previous;
377            }
378    
379            @Override
380            public String toString() {
381                return getKey() + "=" + getValue();
382            }
383        }

552                return CompactHashMap.this.containsValue(o);
553            }
554    
555            @Override
556            public int hashCode() {
557                int result = 0;
558                for (int i = 0; i < keys.length; ++i) {
559                    if (keys[i] != null) {
560                        result += valueHashCode(values[i]);
561                    }

572            public boolean remove(Object o) {
573                if (o == null) {
574                    for (int i = 0; i < keys.length; ++i) {
575                        if (keys[i] != null && values[i] == null) {
576                            internalRemove(i);
577                            return true;
578                        }
579                    }
580                } else {
581                    for (int i = 0; i < keys.length; ++i) {
582                        if (valueEquals(values[i], o)) {
583                            internalRemove(i);
584                            return true;
585                        }
586                    }
587                }
588                return false;
589            }

646        /**
647         * Ensures the map is large enough to contain the specified number of entries.
648         * Default access to avoid synthetic accessors from inner classes.
649         */
650        void ensureSizeFor(int expectedSize) {
651            if (keys.length * 3 >= expectedSize * 4) {
652                return;
653            }
654    
655            int newCapacity = keys.length << 1;
656            while (newCapacity * 3 < expectedSize * 4) {
657                newCapacity <<= 1;
658            }
659    
660            Object[] oldKeys = keys;
661            Object[] oldValues = values;
662            initTable(newCapacity);
663            for (int i = 0; i < oldKeys.length; ++i) {
664                Object k = oldKeys[i];
665                if (k != null) {
666                    int newIndex = getKeyIndex(unmaskNullKey(k));
667                    while (keys[newIndex] != null) {
668                        if (++newIndex == keys.length) {
669                            newIndex = 0;
670                        }
671                    }
672                    keys[newIndex] = k;
673                    values[newIndex] = oldValues[i];
674                }
675            }
676        }
677    
678        /**
679         * Returns the index in the key table at which a particular key resides, or -1
680         * if the key is not in the table. Default access to avoid synthetic accessors
681         * from inner classes.
682         */
683        int findKey(Object k) {
684            int index = getKeyIndex(k);
685            while (true) {
686                Object existing = keys[index];
687                if (existing == null) {
688                    return -1;
689                }
690                if (keyEquals(k, unmaskNullKey(existing))) {
691                    return index;
692                }
693                if (++index == keys.length) {
694                    index = 0;
695                }
696            }
697        }
698    
699        /**
700         * Returns the index in the key table at which a particular key resides, or
701         * the index of an empty slot in the table where this key should be inserted
702         * if it is not already in the table. Default access to avoid synthetic
703         * accessors from inner classes.
704         */
705        int findKeyOrEmpty(Object k) {
706            int index = getKeyIndex(k);
707            while (true) {
708                Object existing = keys[index];
709                if (existing == null) {
710                    return index;
711                }
712                if (keyEquals(k, unmaskNullKey(existing))) {
713                    return index;
714                }
715                if (++index == keys.length) {
716                    index = 0;
717                }
718            }
719        }
720    
721        /**
722         * Removes the entry at the specified index, and performs internal management
723         * to make sure we don't wind up with a hole in the table. Default access to
724         * avoid synthetic accessors from inner classes.
725         */
726        void internalRemove(int index) {
727            keys[index] = null;
728            values[index] = null;
729            --size;
730            plugHole(index);
731        }
732    
733        private int getKeyIndex(Object k) {
734            int h = keyHashCode(k);
735            // Copied from Apache's AbstractHashedMap; prevents power-of-two collisions.
736            h += ~(h << 9);
737            h ^= (h >>> 14);
738            h += (h << 4);
739            h ^= (h >>> 10);
740            // Power of two trick.
741            return h & (keys.length - 1);
742        }
743    
744        private void initTable(int capacity) {
745            keys = new Object[capacity];
746            values = new Object[capacity];
747        }
748    
749        private void internalPutAll(Map<? extends K, ? extends V> m) {
750            for (Entry<? extends K, ? extends V> entry : m.entrySet()) {
751                K key = entry.getKey();
752                V value = entry.getValue();
753                int index = findKeyOrEmpty(key);
754                if (keys[index] == null) {
755                    ++size;
756                    keys[index] = maskNullKey(key);
757                    values[index] = value;
758                } else {
759                    values[index] = value;
760                }
761            }
762        }
763    
764        /**
765         * Tricky, we left a hole in the map, which we have to fill. The only way to
766         * do this is to search forwards through the map shuffling back values that
767         * match this index until we hit a null.
768         */
769        private void plugHole(int hole) {
770            int index = hole + 1;
771            if (index == keys.length) {
772                index = 0;
773            }
774            while (keys[index] != null) {
775                int targetIndex = getKeyIndex(unmaskNullKey(keys[index]));
776                if (hole < index) {
777                    /*
778                     * "Normal" case, the index is past the hole and the "bad range" is from
779                     * hole (exclusive) to index (inclusive).
780                     */
781                    if (!(hole < targetIndex && targetIndex <= index)) {
782                        // Plug it!
783                        keys[hole] = keys[index];
784                        values[hole] = values[index];
785                        keys[index] = null;
786                        values[index] = null;
787                        hole = index;
788                    }
789                } else {
790                    /*
791                     * "Wrapped" case, the index is before the hole (we've wrapped) and the
792                     * "good range" is from index (exclusive) to hole (inclusive).
793                     */
794                    if (index < targetIndex && targetIndex <= hole) {
795                        // Plug it!
796                        keys[hole] = keys[index];
797                        values[hole] = values[index];
798                        keys[index] = null;
799                        values[index] = null;
800                        hole = index;
801                    }
802                }
803                if (++index == keys.length) {
804                    index = 0;
805                }
806            }
807        }
