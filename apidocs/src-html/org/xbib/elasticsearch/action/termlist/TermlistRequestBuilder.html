002package org.xbib.elasticsearch.action.termlist;
003
004import org.elasticsearch.action.ActionListener;
005import org.elasticsearch.action.support.broadcast.BroadcastOperationRequestBuilder;
006import org.elasticsearch.client.Client;
007import org.elasticsearch.client.internal.InternalGenericClient;
008
009/**
010 * A request to get termlists of one or more indices.
011 */
012public class TermlistRequestBuilder extends BroadcastOperationRequestBuilder<TermlistRequest, TermlistResponse, TermlistRequestBuilder> {
013
014    public TermlistRequestBuilder(InternalGenericClient client) {
015        super(client, new TermlistRequest());
016    }
017
018    @Override
019    protected void doExecute(ActionListener<TermlistResponse> listener) {
020        ((Client) client).execute(TermlistAction.INSTANCE, request, listener);
021    }
022}
