002package org.xbib.elasticsearch.action.termlist;
003
004import org.elasticsearch.action.ShardOperationFailedException;
005import org.elasticsearch.action.support.broadcast.BroadcastOperationResponse;
006import org.elasticsearch.common.io.stream.StreamInput;
007import org.elasticsearch.common.io.stream.StreamOutput;
008
009import java.io.IOException;
010import java.util.List;
011import java.util.Map;
012
013/**
014 * A response for termlist action.
015 */
016public class TermlistResponse extends BroadcastOperationResponse {
017
018    private Map<String,TermInfo> map;
019
020    TermlistResponse() {
021    }
022
023    TermlistResponse(int totalShards, int successfulShards, int failedShards,
024                     List<ShardOperationFailedException> shardFailures, Map<String,TermInfo> map) {
025        super(totalShards, successfulShards, failedShards, shardFailures);
026        this.map = map;
027    }
028
029    public Map<String,TermInfo> getTermlist() {
030        return map;
031    }
032
033    @Override
034    public void readFrom(StreamInput in) throws IOException {
035        super.readFrom(in);
036        int n = in.readInt();
037        map = new CompactHashMap<String,TermInfo>();
038        for (int i = 0; i < n; i++) {
039            String text = in.readString();
040            TermInfo t = new TermInfo();
041            t.readFrom(in);
042            map.put(text,t);
043        }
044    }
045
046    @Override
047    public void writeTo(StreamOutput out) throws IOException {
048        super.writeTo(out);
049        out.writeInt(map.size());
050        for (Map.Entry<String,TermInfo> t : map.entrySet()) {
051            out.writeString(t.getKey());
052            t.getValue().writeTo(out);
053        }
054    }
055}
