048    /**
049     * Compares two numbers given some amount of allowed error.
050     *
051     * @param x the first number
052     * @param y the second number
053     * @param eps the amount of error to allow when checking for equality
054     * @return <ul>
055     *     <li>0 if  {@link #equals(double, double, double) equals(x, y, eps)}</li>
056     *     <li>&lt; 0 if !{@link #equals(double, double, double) equals(x, y, eps)} &amp;&amp; x &lt; y</li>
057     *     <li>&gt; 0 if !{@link #equals(double, double, double) equals(x, y, eps)} &amp;&amp; x &gt; y</li>
058     *     </ul>
059     */
060    public static int compareTo(double x, double y, double eps) {
061        if (equals(x, y, eps)) {
062            return 0;
063        } else if (x < y) {
064            return -1;
065        }
066        return 1;
067    }

069    /**
070     * Compares two numbers given some amount of allowed error.
071     * Two float numbers are considered equal if there are {@code (maxUlps - 1)}
072     * (or fewer) floating point numbers between them, i.e. two adjacent floating
073     * point numbers are considered equal.
074     * Adapted from <a
075     * href="http://www.cygnus-software.com/papers/comparingfloats/comparingfloats.htm">
076     * Bruce Dawson</a>
077     *
078     * @param x first value
079     * @param y second value
080     * @param maxUlps {@code (maxUlps - 1)} is the number of floating point
081     * values between {@code x} and {@code y}.
082     * @return <ul>
083     *     <li>0 if  {@link #equals(double, double, int) equals(x, y, maxUlps)}</li>
084     *     <li>&lt; 0 if !{@link #equals(double, double, int) equals(x, y, maxUlps)} &amp;&amp; x &lt; y</li>
085     *     <li>&gt; 0 if !{@link #equals(double, double, int) equals(x, y, maxUlps)} &amp;&amp; x &gt; y</li>
086     *     </ul>
087     */
088    public static int compareTo(final double x, final double y, final int maxUlps) {
089        if (equals(x, y, maxUlps)) {
090            return 0;
091        } else if (x < y) {
092            return -1;
093        }
094        return 1;
095    }

148    /**
149     * Returns true if both arguments are equal or within the range of allowed
150     * error (inclusive).
151     * Two float numbers are considered equal if there are {@code (maxUlps - 1)}
152     * (or fewer) floating point numbers between them, i.e. two adjacent floating
153     * point numbers are considered equal.
154     * Adapted from <a
155     * href="http://www.cygnus-software.com/papers/comparingfloats/comparingfloats.htm">
156     * Bruce Dawson</a>
157     *
158     * @param x first value
159     * @param y second value
160     * @param maxUlps {@code (maxUlps - 1)} is the number of floating point
161     * values between {@code x} and {@code y}.
162     * @return {@code true} if there are fewer than {@code maxUlps} floating
163     * point values between {@code x} and {@code y}.
164     */
165    public static boolean equals(float x, float y, int maxUlps) {
166        int xInt = Float.floatToIntBits(x);
167        int yInt = Float.floatToIntBits(y);
168
169        // Make lexicographically ordered as a two's-complement integer.
170        if (xInt < 0) {
171            xInt = SGN_MASK_FLOAT - xInt;
172        }
173        if (yInt < 0) {
174            yInt = SGN_MASK_FLOAT - yInt;
175        }
176
177        final boolean isEqual = FastMath.abs(xInt - yInt) <= maxUlps;
178
179        return isEqual && !Float.isNaN(x) && !Float.isNaN(y);
180    }

311    /**
312     * Rounds the given value to the specified number of decimal places.
313     * The value is rounded using the given method which is any method defined
314     * in {@link BigDecimal}.
315     * If {@code x} is infinite or {@code NaN}, then the value of {@code x} is
316     * returned unchanged, regardless of the other parameters.
317     *
318     * @param x Value to round.
319     * @param scale Number of digits to the right of the decimal point.
320     * @param roundingMethod Rounding method as defined in {@link BigDecimal}.
321     * @return the rounded value.
322     * @throws ArithmeticException if {@code roundingMethod == ROUND_UNNECESSARY}
323     * and the specified scaling operation would require rounding.
324     * @throws IllegalArgumentException if {@code roundingMethod} does not
325     * represent a valid rounding mode.
326     */
327    public static double round(double x, int scale, int roundingMethod) {
328        try {
329            return (new BigDecimal
330                   (Double.toString(x))
331                   .setScale(scale, roundingMethod))
332                   .doubleValue();
333        } catch (NumberFormatException ex) {
334            if (Double.isInfinite(x)) {
335                return x;
336            } else {
337                return Double.NaN;
338            }
339        }
340    }

354    /**
355     * Rounds the given value to the specified number of decimal places.
356     * The value is rounded using the given method which is any method defined
357     * in {@link BigDecimal}.
358     *
359     * @param x Value to round.
360     * @param scale Number of digits to the right of the decimal point.
361     * @param roundingMethod Rounding method as defined in {@link BigDecimal}.
362     * @return the rounded value.
363     */
364    public static float round(float x, int scale, int roundingMethod) {
365        final float sign = FastMath.copySign(1f, x);
366        final float factor = (float) FastMath.pow(10.0f, scale) * sign;
367        return (float) roundUnscaled(x * factor, sign, roundingMethod) / factor;
368    }

370    /**
371     * Rounds the given non-negative value to the "nearest" integer. Nearest is
372     * determined by the rounding method specified. Rounding methods are defined
373     * in {@link BigDecimal}.
374     *
375     * @param unscaled Value to round.
376     * @param sign Sign of the original, scaled value.
377     * @param roundingMethod Rounding method, as defined in {@link BigDecimal}.
378     * @return the rounded value.
379     */
380    private static double roundUnscaled(double unscaled,
381                                        double sign,
382                                        int roundingMethod) {
383        switch (roundingMethod) {
384        case BigDecimal.ROUND_CEILING :
385            if (sign == -1) {
386                unscaled = FastMath.floor(FastMath.nextAfter(unscaled, Double.NEGATIVE_INFINITY));
387            } else {
388                unscaled = FastMath.ceil(FastMath.nextAfter(unscaled, Double.POSITIVE_INFINITY));
389            }
390            break;
391        case BigDecimal.ROUND_DOWN :
392            unscaled = FastMath.floor(FastMath.nextAfter(unscaled, Double.NEGATIVE_INFINITY));
393            break;
394        case BigDecimal.ROUND_FLOOR :
395            if (sign == -1) {
396                unscaled = FastMath.ceil(FastMath.nextAfter(unscaled, Double.POSITIVE_INFINITY));
397            } else {
398                unscaled = FastMath.floor(FastMath.nextAfter(unscaled, Double.NEGATIVE_INFINITY));
399            }
400            break;
401        case BigDecimal.ROUND_HALF_DOWN : {
402            unscaled = FastMath.nextAfter(unscaled, Double.NEGATIVE_INFINITY);
403            double fraction = unscaled - FastMath.floor(unscaled);
404            if (fraction > 0.5) {
405                unscaled = FastMath.ceil(unscaled);
406            } else {
407                unscaled = FastMath.floor(unscaled);
408            }
409            break;
410        }
411        case BigDecimal.ROUND_HALF_EVEN : {
412            double fraction = unscaled - FastMath.floor(unscaled);
413            if (fraction > 0.5) {
414                unscaled = FastMath.ceil(unscaled);
415            } else if (fraction < 0.5) {
416                unscaled = FastMath.floor(unscaled);
417            } else {
418                // The following equality test is intentional and needed for rounding purposes
419                if (FastMath.floor(unscaled) / 2.0 == FastMath.floor(Math
420                    .floor(unscaled) / 2.0)) { // even
421                    unscaled = FastMath.floor(unscaled);
422                } else { // odd
423                    unscaled = FastMath.ceil(unscaled);
424                }
425            }
426            break;
427        }
428        case BigDecimal.ROUND_HALF_UP : {
429            unscaled = FastMath.nextAfter(unscaled, Double.POSITIVE_INFINITY);
430            double fraction = unscaled - FastMath.floor(unscaled);
431            if (fraction >= 0.5) {
432                unscaled = FastMath.ceil(unscaled);
433            } else {
434                unscaled = FastMath.floor(unscaled);
435            }
436            break;
437        }
438        case BigDecimal.ROUND_UNNECESSARY :
439            if (unscaled != FastMath.floor(unscaled)) {
440                throw new IllegalArgumentException();
441            }
442            break;
443        case BigDecimal.ROUND_UP :
444            unscaled = FastMath.ceil(FastMath.nextAfter(unscaled,  Double.POSITIVE_INFINITY));
445            break;
446        default :
447            throw new IllegalArgumentException("invalid rounding");
448        }
449        return unscaled;
450    }

453    /**
454     * Computes a number {@code delta} close to {@code originalDelta} with
455     * the property that <pre><code>
456     *   x + delta - x
457     * </code></pre>
458     * is exactly machine-representable.
459     * This is useful when computing numerical derivatives, in order to reduce
460     * roundoff errors.
461     *
462     * @param x Value.
463     * @param originalDelta Offset value.
464     * @return a number {@code delta} so that {@code x + delta} and {@code x}
465     * differ by a representable floating number.
466     */
467    public static double representableDelta(double x,
468                                            double originalDelta) {
469        return x + originalDelta - x;
470    }
