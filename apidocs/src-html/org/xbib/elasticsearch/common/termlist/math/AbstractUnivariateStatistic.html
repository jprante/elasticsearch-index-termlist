017package org.xbib.elasticsearch.common.termlist.math;
018
019import org.elasticsearch.common.io.stream.Streamable;
020
021/**
022 * Abstract base class for all implementations of the
023 * {@link UnivariateStatistic} interface.
024 *
025 * Provides a default implementation of <code>evaluate(double[]),</code>
026 * delegating to <code>evaluate(double[], int, int)</code> in the natural way.
027 *
028 *
029 * Also includes a <code>test</code> method that performs generic parameter
030 * validation for the <code>evaluate</code> methods.
031 *
032 */
033public abstract class AbstractUnivariateStatistic
034    implements UnivariateStatistic, Streamable {
035
036    /** Stored data. */
037    private double[] storedData;
038
039    /**
040     * Set the data array.
041     *
042     * The stored value is a copy of the parameter array, not the array itself
043     *
044     * @param values data array to store (may be null to remove stored data)
045     * @see #evaluate()
046     */
047    public void setData(final double[] values) {
048        storedData = (values == null) ? null : values.clone();
049    }
050
051    /**
052     * Get a copy of the stored data array.
053     * @return copy of the stored data array (may be null)
054     */
055    public double[] getData() {
056        return (storedData == null) ? null : storedData.clone();
057    }
058
059    /**
060     * Get a reference to the stored data array.
061     * @return reference to the stored data array (may be null)
062     */
063    protected double[] getDataRef() {
064        return storedData;
065    }
066
067    /**
068     * Set the data array.
069     * @param values data array to store
070     * @param begin the index of the first element to include
071     * @param length the number of elements to include
072     * @see #evaluate()
073     */
074    public void setData(final double[] values, final int begin, final int length) {
075        storedData = new double[length];
076        System.arraycopy(values, begin, storedData, 0, length);
077    }
078
079    /**
080     * Returns the result of evaluating the statistic over the stored data.
081     * The stored array is the one which was set by previous calls to
082     * @return the value of the statistic applied to the stored data
083     */
084    public double evaluate() {
085        return evaluate(storedData);
086    }
087
088    public double evaluate(final double[] values) {
089        test(values, 0, 0);
090        return evaluate(values, 0, values.length);
091    }
092
093    public abstract double evaluate(final double[] values, final int begin, final int length);
094
095    /**
096     * This method is used by <code>evaluate(double[], int, int)</code> methods
097     * to verify that the input parameters designate a subarray of positive length.
098     * <ul>
099     * <li>returns <code>true</code> iff the parameters designate a subarray of
100     * positive length</li>
101     * <li>throws <code>IllegalArgumentException</code> if the array is null or
102     * or the indices are invalid</li>
103     * <li>returns <code>false</code> if the array is non-null, but
104     * <code>length</code> is 0.</li>
105     * </ul>
106     *
107     * @param values the input array
108     * @param begin index of the first array element to include
109     * @param length the number of elements to include
110     * @return true if the parameters are valid and designate a subarray of positive length
111     * @throws IllegalArgumentException if the indices are invalid or the array is null
112     */
113    protected boolean test(
114        final double[] values,
115        final int begin,
116        final int length) {
117        return test(values, begin, length, false);
118    }
119
120    /**
121     * This method is used by <code>evaluate(double[], int, int)</code> methods
122     * to verify that the input parameters designate a subarray of positive length.
123     *
124     * <ul>
125     * <li>returns <code>true</code> iff the parameters designate a subarray of
126     * non-negative length</li>
127     * <li>throws <code>IllegalArgumentException</code> if the array is null or
128     * or the indices are invalid</li>
129     * <li>returns <code>false</code> if the array is non-null, but
130     * <code>length</code> is 0 unless <code>allowEmpty</code> is <code>true</code></li>
131     * </ul>
132     *
133     * @param values the input array
134     * @param begin index of the first array element to include
135     * @param length the number of elements to include
136     * @param allowEmpty if <code>true</code> then zero length arrays are allowed
137     * @return true if the parameters are valid
138     * @throws IllegalArgumentException if the indices are invalid or the array is null
139     */
140    protected boolean test(final double[] values, final int begin, final int length, final boolean allowEmpty){
141        if (values == null) {
142            throw new IllegalArgumentException("values is null");
143        }
144        if (begin < 0) {
145            throw new IllegalArgumentException("begin < 0");
146        }
147        if (length < 0) {
148            throw new IllegalArgumentException("length < 0");
149        }
150        if (begin + length > values.length) {
151            throw new IllegalArgumentException("begin + length > values.length");
152        }
153        if (length == 0 && !allowEmpty) {
154            return false;
155        }
156        return true;
157
158    }
159
160    /**
161     * This method is used by <code>evaluate(double[], double[], int, int)</code> methods
162     * to verify that the begin and length parameters designate a subarray of positive length
163     * and the weights are all non-negative, non-NaN, finite, and not all zero.
164     *
165     * <ul>
166     * <li>returns <code>true</code> iff the parameters designate a subarray of
167     * positive length and the weights array contains legitimate values.</li>
168     * <li>throws <code>IllegalArgumentException</code> if any of the following are true:
169     * <ul><li>the values array is null</li>
170     *     <li>the weights array is null</li>
171     *     <li>the weights array does not have the same length as the values array</li>
172     *     <li>the weights array contains one or more infinite values</li>
173     *     <li>the weights array contains one or more NaN values</li>
174     *     <li>the weights array contains negative values</li>
175     *     <li>the start and length arguments do not determine a valid array</li></ul>
176     * </li>
177     * <li>returns <code>false</code> if the array is non-null, but
178     * <code>length</code> is 0.</li>
179     * </ul>
180     *
181     * @param values the input array
182     * @param weights the weights array
183     * @param begin index of the first array element to include
184     * @param length the number of elements to include
185     * @return true if the parameters are valid and designate a subarray of positive length
186     * @throws IllegalArgumentException if the indices are invalid or the array is null
187     */
188    protected boolean test(
189        final double[] values,
190        final double[] weights,
191        final int begin,
192        final int length) {
193        return test(values, weights, begin, length, false);
194    }
195
196    /**
197     * This method is used by <code>evaluate(double[], double[], int, int)</code> methods
198     * to verify that the begin and length parameters designate a subarray of positive length
199     * and the weights are all non-negative, non-NaN, finite, and not all zero.
200     *
201     * <ul>
202     * <li>returns <code>true</code> iff the parameters designate a subarray of
203     * non-negative length and the weights array contains legitimate values.</li>
204     * <li>throws <code>IllegalArgumentException</code> if any of the following are true:
205     * <ul><li>the values array is null</li>
206     *     <li>the weights array is null</li>
207     *     <li>the weights array does not have the same length as the values array</li>
208     *     <li>the weights array contains one or more infinite values</li>
209     *     <li>the weights array contains one or more NaN values</li>
210     *     <li>the weights array contains negative values</li>
211     *     <li>the start and length arguments do not determine a valid array</li></ul>
212     * </li>
213     * <li>returns <code>false</code> if the array is non-null, but
214     * <code>length</code> is 0 unless <code>allowEmpty</code> is <code>true</code>.</li>
215     * </ul>
216     *
217     * @param values the input array.
218     * @param weights the weights array.
219     * @param begin index of the first array element to include.
220     * @param length the number of elements to include.
221     * @param allowEmpty if {@code true} than allow zero length arrays to pass.
222     * @return {@code true} if the parameters are valid.
223     * @throws IllegalArgumentException if the indices are invalid or the array
224     * is {@code null}.
225     */
226    protected boolean test(final double[] values, final double[] weights, final int begin, final int length, final boolean allowEmpty){
227
228        if (weights == null) {
229            throw new IllegalArgumentException("weights == null");
230        }
231
232        if (weights.length != values.length) {
233            throw new IllegalArgumentException("weights.length != values.length");
234        }
235
236        boolean containsPositiveWeight = false;
237        for (int i = begin; i < begin + length; i++) {
238            if (Double.isNaN(weights[i])) {
239                throw new IllegalArgumentException("Double.isNaN(weights[i])");
240            }
241            if (Double.isInfinite(weights[i])) {
242                throw new IllegalArgumentException("Double.isInfinite(weights[i])");
243            }
244            if (weights[i] < 0) {
245                throw new IllegalArgumentException("weights[i] < 0");
246            }
247            if (!containsPositiveWeight && weights[i] > 0.0) {
248                containsPositiveWeight = true;
249            }
250        }
251
252        if (!containsPositiveWeight) {
253            throw new IllegalArgumentException("containsPositiveWeight");
254        }
255
256        return test(values, begin, length, allowEmpty);
257    }
258}
