017package org.xbib.elasticsearch.common.termlist.math;
018
019/**
020 * Provides default <code>evaluate()</code> and <code>incrementAll(double[])</code>
021 * implementations.
022 *
023 * <strong>Note that these implementations are not synchronized.</strong>
024 */
025public abstract class AbstractStorelessUnivariateStatistic
026    extends AbstractUnivariateStatistic
027    implements StorelessUnivariateStatistic {
028
029    /**
030     * This default implementation calls {@link #clear}, then invokes
031     * {@link #increment} in a loop over the the input array, and then uses
032     * {@link #getResult} to compute the return value.
033     *
034     * Note that this implementation changes the internal state of the
035     * statistic.  Its side effects are the same as invoking {@link #clear} and
036     * then {@link #incrementAll(double[])}.
037     *
038     * Implementations may override this method with a more efficient and
039     * possibly more accurate implementation that works directly with the
040     * input array.
041     *
042     * If the array is null, an IllegalArgumentException is thrown.
043     * @param values input array
044     * @return the value of the statistic applied to the input array
045     */
046    @Override
047    public double evaluate(final double[] values) {
048        if (values == null) {
049            throw new IllegalArgumentException();
050        }
051        return evaluate(values, 0, values.length);
052    }
053
054    /**
055     * This default implementation calls {@link #clear}, then invokes
056     * {@link #increment} in a loop over the specified portion of the input
057     * array, and then uses {@link #getResult} to compute the return value.
058     *
059     * Note that this implementation changes the internal state of the
060     * statistic.  Its side effects are the same as invoking {@link #clear} and
061     * then {@link #incrementAll(double[], int, int)}.
062     *
063     * Implementations may override this method with a more efficient and
064     * possibly more accurate implementation that works directly with the
065     * input array.
066     *
067     * If the array is null or the index parameters are not valid, an
068     * IllegalArgumentException is thrown.
069     * @param values the input array
070     * @param begin the index of the first element to include
071     * @param length the number of elements to include
072     * @return the value of the statistic applied to the included array entries
073     */
074    @Override
075    public double evaluate(final double[] values, final int begin, final int length) {
076        if (test(values, begin, length)) {
077            clear();
078            incrementAll(values, begin, length);
079        }
080        return getResult();
081    }
082
083    public abstract void clear();
084
085    public abstract double getResult();
086
087    public abstract void increment(final double d);
088
089    /**
090     * This default implementation just calls {@link #increment} in a loop over
091     * the input array.
092     * <p>
093     * Throws IllegalArgumentException if the input values array is null.</p>
094     *
095     * @param values values to add
096     * @throws IllegalArgumentException if values is null
097     */
098    public void incrementAll(double[] values) {
099        if (values == null) {
100            throw new IllegalArgumentException();
101        }
102        incrementAll(values, 0, values.length);
103    }
104
105    /**
106     * This default implementation just calls {@link #increment} in a loop over
107     * the specified portion of the input array.
108     * <p>
109     * Throws IllegalArgumentException if the input values array is null.</p>
110     *
111     * @param values  array holding values to add
112     * @param begin   index of the first array element to add
113     * @param length  number of array elements to add
114     * @throws IllegalArgumentException if values is null
115     */
116    public void incrementAll(double[] values, int begin, int length) {
117        if (test(values, begin, length)) {
118            int k = begin + length;
119            for (int i = begin; i < k; i++) {
120                increment(values[i]);
121            }
122        }
123    }
124
125    /**
126     * Returns true iff <code>object</code> is an
127     * <code>AbstractStorelessUnivariateStatistic</code> returning the same
128     * values as this for <code>getResult()</code> and <code>getN()</code>
129     * @param object object to test equality against.
130     * @return true if object returns the same value as this
131     */
132    @Override
133    public boolean equals(Object object) {
134        if (object == this ) {
135            return true;
136        }
137       if (!(object instanceof AbstractStorelessUnivariateStatistic)) {
138            return false;
139        }
140        AbstractStorelessUnivariateStatistic stat = (AbstractStorelessUnivariateStatistic) object;
141        return Precision.equalsIncludingNaN(stat.getResult(), this.getResult()) &&
142               Precision.equalsIncludingNaN(stat.getN(), this.getN());
143    }
144
145    /**
146     * Returns hash code based on getResult() and getN()
147     *
148     * @return hash code
149     */
150    @Override
151    public int hashCode() {
152        return 31* (31 + new Double(getResult()).hashCode()) + new Double(getN()).hashCode();
153    }
154
155}
